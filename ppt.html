Guia de Programação: Pedra, Papel ou Tesoura

Bloco 1: Início e Variáveis

import random # 1. Definir as jogadas jogadas = ["pedra", "papel", "tesoura"] print() print("--- JOGO: PEDRA, PAPEL OU TESOURA ---")

Bloco 2: O Ciclo de Escolha (Loop)

while True: print("Podes jogar: pedra, papel ou tesoura.") jogada_utilizador = input("Escreve a tua escolha: ") # passar para minúsculas jogada_utilizador = jogada_utilizador.lower() if jogada_utilizador in jogadas: break else: print("Escolha inválida. Tenta novamente.")

Bloco 3: A Lógica do Computador

jogada_computador = random.choice(jogadas) print() print("Tu escolheste:", jogada_utilizador) print("O computador escolheu:", jogada_computador)

Bloco 4: Condições de Vitória

if jogada_utilizador == jogada_computador: print("Resultado: Empataram!") elif jogada_utilizador == "pedra" and jogada_computador == "tesoura": print("Resultado: Tu ganhaste!") elif jogada_utilizador == "papel" and jogada_computador == "pedra": print("Resultado: Tu ganhaste!") elif jogada_utilizador == "tesoura" and jogada_computador == "papel": print("Resultado: Tu ganhaste!") else: print("Resultado: O computador ganhou!")
