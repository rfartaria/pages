import random
jogadas = ["pedra", "papel", "tesoura"]
jog_comp = random.choice(jogadas)

while True:
    print("Podes escolher: pedra, papel ou tesoura.")
    jog_hum = input("Qual escolhes?")
    if jog_hum in jogadas:
        break
    else:
        print(f"{jog_hum} não é uma jogada válida. Tenta outra vez.")
        print()

print(f"Eu joguei {jog_comp} e tu jogaste {jog_hum}.")
